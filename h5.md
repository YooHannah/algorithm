//实现把A标签拖到B标签
1.A标签开始拖动时
ondragstart(event){
  event.dataTransfer.setData('source',event.target.id)
}
2.B标签监听拖拽结束后
ondrop(event){
  var source = document.getElementById(event.dataTransfer.getData('source'))
  event.target.appendChild(source)
}
